``` r

  filter(year == 2007) %>% 
  mutate(GDP = gdpPercap * pop)


gdp_pop_lifeExp <- ggplot(gap_filtered, aes(GDP, pop)) +
    geom_point(aes(colour=lifeExp)) + 
    scale_x_log10(labels= dollar_format(), breaks = 10^((1:20)) ) +
    scale_y_log10(labels = comma_format()) +
```
